```python
import re
from typing import Any, List, Tuple

def parse(query: str) -> List[Tuple[str, str, Any]]:
    """Parse query string into list of (field, op, value) predicates."""
    predicates = []
    conditions = query.split(' AND ')

    for condition in conditions:
        condition = condition.strip()
        match_obj = re.match(r"(\w+)\s*(>=|<=|!=|=|>|<)\s*(.+)", condition)
        if match_obj:
            field, op, value_str = match_obj.groups()
            value_str = value_str.strip()

            if (value_str.startswith("'") and value_str.endswith("'")) or \
               (value_str.startswith('"') and value_str.endswith('"')):
                value = value_str[1:-1]
            else:
                try:
                    value = float(value_str) if '.' in value_str else int(value_str)
                except ValueError:
                    value = value_str

            predicates.append((field, op, value))

    return predicates
```

```python
from typing import Any, Dict, List, Tuple

def match(row: Dict[str, Any], predicates: List[Tuple[str, str, Any]]) -> bool:
    """Check if row satisfies all predicates."""
    for field, op, value in predicates:
        if field not in row:
            return False

        row_val = row[field]

        if op == '=' and row_val != value:
            return False
        elif op == '!=' and row_val == value:
            return False
        elif op == '>' and not (row_val > value):
            return False
        elif op == '>=' and not (row_val >= value):
            return False
        elif op == '<' and not (row_val < value):
            return False
        elif op == '<=' and not (row_val <= value):
            return False

    return True
```

```python
from typing import Any, Dict, List

def select(rows: List[Dict[str, Any]], query: str) -> List[Dict[str, Any]]:
    """Return rows matching the query."""
    from __main__ import parse, match
    predicates = parse(query)
    return [row for row in rows if match(row, predicates)]
```

```python
from typing import Any, Dict, List, Union

def summarize(rows: List[Dict[str, Any]], field: str, op: str) -> Union[int, float]:
    """Aggregate field with operation (count/sum/min/max/avg)."""
    if op == 'count':
        return len(rows)

    values = [row[field] for row in rows if field in row]

    if not values:
        return 0

    if op == 'sum':
        return sum(values)
    elif op == 'min':
        return min(values)
    elif op == 'max':
        return max(values)
    elif op == 'avg':
        return sum(values) / len(values)

    return 0
```

```python
from typing import Any, Dict, List

def render(rows: List[Dict[str, Any]], columns: List[str]) -> str:
    """Format rows as a simple text table."""
    if not rows:
        return ""

    widths = {}
    for col in columns:
        widths[col] = len(col)
        for row in rows:
            if col in row:
                widths[col] = max(widths[col], len(str(row[col])))

    lines = []
    header = " | ".join(col.ljust(widths[col]) for col in columns)
    lines.append(header)
    lines.append("-" * len(header))

    for row in rows:
        row_str = " | ".join(str(row.get(col, "")).ljust(widths[col]) for col in columns)
        lines.append(row_str)

    return "\n".join(lines)
```
