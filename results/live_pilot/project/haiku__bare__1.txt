```python
def parse(query):
    import re
    conditions = []
    parts = re.split(r'\s+AND\s+', query)
    for part in parts:
        part = part.strip()
        match = re.match(r"(\w+)\s*(<=|>=|!=|=|<|>)\s*(.+)", part)
        if match:
            field, op, value_str = match.groups()
            value_str = value_str.strip()
            # Parse value: try as number, else as string (strip quotes)
            if (value_str.startswith("'") and value_str.endswith("'")) or \
               (value_str.startswith('"') and value_str.endswith('"')):
                value = value_str[1:-1]
            else:
                try:
                    value = int(value_str)
                except ValueError:
                    try:
                        value = float(value_str)
                    except ValueError:
                        value = value_str
            conditions.append({'field': field, 'op': op, 'value': value})
    return conditions
```

```python
def match(row, predicates):
    for pred in predicates:
        field = pred['field']
        op = pred['op']
        value = pred['value']

        if field not in row:
            return False

        row_value = row[field]

        if op == '=' and row_value != value:
            return False
        elif op == '!=' and row_value == value:
            return False
        elif op == '<' and not (row_value < value):
            return False
        elif op == '>' and not (row_value > value):
            return False
        elif op == '<=' and not (row_value <= value):
            return False
        elif op == '>=' and not (row_value >= value):
            return False

    return True
```

### MODULE: select
CONFIDENCE: 0.95

```python
def select(rows, query):
    predicates = parse(query)
    return [row for row in rows if match(row, predicates)]
```

```python
def summarize(rows, field, op):
    if op == 'count':
        return len(rows)

    values = [row[field] for row in rows if field in row]

    if op == 'sum':
        return sum(values) if values else 0
    elif op == 'min':
        return min(values) if values else None
    elif op == 'max':
        return max(values) if values else None
    elif op == 'avg':
        return sum(values) / len(values) if values else None
```

```python
def render(rows, columns):
    if not rows:
        return ""

    # Calculate column widths
    widths = {}
    for col in columns:
        widths[col] = len(str(col))
        for row in rows:
            if col in row:
                widths[col] = max(widths[col], len(str(row[col])))

    # Build header and separator
    header = " | ".join(str(col).ljust(widths[col]) for col in columns)
    separator = "-+-".join("-" * widths[col] for col in columns)

    # Build rows
    result = [header, separator]
    for row in rows:
        row_str = " | ".join(str(row.get(col, "")).ljust(widths[col]) for col in columns)
        result.append(row_str)

    return "\n".join(result)
```
