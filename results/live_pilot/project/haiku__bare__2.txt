def parse(query):
    """Parse a query string into a list of conditions."""
    if not query.strip():
        return []

    conditions = []
    parts = query.split(" AND ")

    for part in parts:
        part = part.strip()
        # Try to match operators in order (longest first)
        for op in [">=", "<=", "!=", "=", ">", "<"]:
            if op in part:
                field, value = part.split(op, 1)
                field = field.strip()
                value = value.strip()

                # Parse value: remove quotes if present, else try numeric conversion
                if value.startswith("'") and value.endswith("'"):
                    value = value[1:-1]
                elif value.startswith('"') and value.endswith('"'):
                    value = value[1:-1]
                else:
                    try:
                        if '.' in value:
                            value = float(value)
                        else:
                            value = int(value)
                    except ValueError:
                        pass

                conditions.append((field, op, value))
                break

    return conditions
```

def match(row, predicates):
    """Check if a row satisfies all predicates."""
    for field, op, value in predicates:
        if field not in row:
            return False

        row_val = row[field]

        if op == "=":
            if row_val != value:
                return False
        elif op == "!=":
            if row_val == value:
                return False
        elif op == ">":
            if row_val <= value:
                return False
        elif op == ">=":
            if row_val < value:
                return False
        elif op == "<":
            if row_val >= value:
                return False
        elif op == "<=":
            if row_val > value:
                return False

    return True
```

CONFIDENCE: 0.85
```python
def select(rows, query):
    """Return rows matching the query."""
    predicates = parse(query)
    return [row for row in rows if match(row, predicates)]
```

def summarize(rows, field, op):
    """Aggregate a field with op in count/sum/min/max/avg."""
    if not rows:
        return None

    values = [row[field] for row in rows if field in row]

    if op == "count":
        return len(values)
    elif op == "sum":
        return sum(values)
    elif op == "min":
        return min(values) if values else None
    elif op == "max":
        return max(values) if values else None
    elif op == "avg":
        return sum(values) / len(values) if values else None

    return None
```

def render(rows, columns):
    """Format rows as a simple text table."""
    if not rows or not columns:
        return ""

    # Calculate column widths
    widths = {}
    for col in columns:
        widths[col] = len(str(col))
        for row in rows:
            widths[col] = max(widths[col], len(str(row.get(col, ""))))

    # Build header
    lines = []
    header_parts = [str(col).ljust(widths[col]) for col in columns]
    header = " | ".join(header_parts)
    lines.append(header)

    # Build separator
    sep_parts = ["-" * widths[col] for col in columns]
    separator = "-+-".join(sep_parts)
    lines.append(separator)

    # Build rows
    for row in rows:
        row_parts = [str(row.get(col, "")).ljust(widths[col]) for col in columns]
        line = " | ".join(row_parts)
        lines.append(line)

    return "\n".join(lines)
```
