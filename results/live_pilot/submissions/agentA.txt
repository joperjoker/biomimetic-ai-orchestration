```python
def run_length_encode(s: str) -> str:
    if not s:
        return ''
    result = []
    prev = s[0]
    count = 1
    for ch in s[1:]:
        if ch == prev:
            count += 1
        else:
            result.append(prev + str(count))
            prev = ch
            count = 1
    result.append(prev + str(count))
    return ''.join(result)
```

```python
def roman_to_int(s: str) -> int:
    values = {'I': 1, 'V': 5, 'X': 10, 'L': 50,
              'C': 100, 'D': 500, 'M': 1000}
    total = 0
    prev = 0
    for ch in reversed(s):
        v = values[ch]
        if v < prev:
            total -= v
        else:
            total += v
            prev = v
    return total
```

```python
def is_balanced(s: str) -> bool:
    pairs = {')': '(', ']': '[', '}': '{'}
    opens = set(pairs.values())
    stack = []
    for ch in s:
        if ch in opens:
            stack.append(ch)
        elif ch in pairs:
            if not stack or stack.pop() != pairs[ch]:
                return False
    return not stack
```

```python
def merge_intervals(intervals: list[list[int]]) -> list[list[int]]:
    if not intervals:
        return []
    ordered = sorted(intervals, key=lambda iv: (iv[0], iv[1]))
    result = [list(ordered[0])]
    for start, end in ordered[1:]:
        last = result[-1]
        if start <= last[1]:
            if end > last[1]:
                last[1] = end
        else:
            result.append([start, end])
    return result
```

```python
def expand_ranges(s: str) -> list[int]:
    result = []
    if not s.strip():
        return result
    for part in s.split(','):
        part = part.strip()
        if not part:
            continue
        if '-' in part[1:]:
            idx = part.index('-', 1)
            a = int(part[:idx])
            b = int(part[idx + 1:])
            if a <= b:
                result.extend(range(a, b + 1))
            else:
                result.extend(range(a, b - 1, -1))
        else:
            result.append(int(part))
    return result
```

```python
def chunk(lst: list, n: int) -> list[list]:
    if n <= 0:
        return []
    return [lst[i:i + n] for i in range(0, len(lst), n)]
```

```python
def deep_get(d: dict, path: str, default=None):
    current = d
    for key in path.split('.'):
        if isinstance(current, dict) and key in current:
            current = current[key]
        else:
            return default
    return current
```

```python
def caesar(s: str, k: int) -> str:
    result = []
    for ch in s:
        if 'a' <= ch <= 'z':
            result.append(chr((ord(ch) - ord('a') + k) % 26 + ord('a')))
        elif 'A' <= ch <= 'Z':
            result.append(chr((ord(ch) - ord('A') + k) % 26 + ord('A')))
        else:
            result.append(ch)
    return ''.join(result)
```

```python
def int_to_roman(n: int) -> str:
    vals = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1]
    syms = ['M', 'CM', 'D', 'CD', 'C', 'XC', 'L', 'XL',
            'X', 'IX', 'V', 'IV', 'I']
    result = []
    for v, sym in zip(vals, syms):
        while n >= v:
            result.append(sym)
            n -= v
    return ''.join(result)
```

```python
def valid_ipv4(s: str) -> bool:
    parts = s.split('.')
    if len(parts) != 4:
        return False
    for part in parts:
        if not part.isdigit():
            return False
        if not part.isascii():
            return False
        if len(part) > 1 and part[0] == '0':
            return False
        if int(part) > 255:
            return False
    return True
```

```python
def next_bigger(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    i = length - 2
    while i >= 0 and digits[i] >= digits[i + 1]:
        i -= 1
    if i < 0:
        return -1
    j = length - 1
    while digits[j] <= digits[i]:
        j -= 1
    digits[i], digits[j] = digits[j], digits[i]
    digits[i + 1:] = reversed(digits[i + 1:])
    return int(''.join(digits))
```

```python
def format_duration(seconds: int) -> str:
    if seconds == 0:
        return 'now'
    units = [
        ('year', 365 * 24 * 3600),
        ('day', 24 * 3600),
        ('hour', 3600),
        ('minute', 60),
        ('second', 1),
    ]
    parts = []
    remaining = seconds
    for name, size in units:
        value, remaining = divmod(remaining, size)
        if value:
            parts.append(f"{value} {name}" + ('s' if value != 1 else ''))
    if len(parts) == 1:
        return parts[0]
    return ', '.join(parts[:-1]) + ' and ' + parts[-1]
```
