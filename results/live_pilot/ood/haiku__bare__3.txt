```python
def my_atoi(s: str) -> int:
    s = s.lstrip()
    if not s:
        return 0

    sign = 1
    idx = 0

    if s[0] in ['+', '-']:
        if s[0] == '-':
            sign = -1
        idx = 1

    result = 0
    while idx < len(s) and s[idx].isdigit():
        result = result * 10 + int(s[idx])
        idx += 1

    result *= sign

    INT_MAX = 2**31 - 1
    INT_MIN = -2**31

    if result > INT_MAX:
        return INT_MAX
    if result < INT_MIN:
        return INT_MIN

    return result
```

```python
def reverse_integer(x: int) -> int:
    INT_MAX = 2**31 - 1
    INT_MIN = -2**31

    sign = 1 if x >= 0 else -1
    x = abs(x)

    result = 0
    while x > 0:
        result = result * 10 + x % 10
        x //= 10

    result *= sign

    if result > INT_MAX or result < INT_MIN:
        return 0

    return result
```

```python
def simplify_path(path: str) -> str:
    parts = path.split('/')
    stack = []

    for part in parts:
        if part == '' or part == '.':
            continue
        elif part == '..':
            if stack:
                stack.pop()
        else:
            stack.append(part)

    return '/' + '/'.join(stack)
```

```python
def compare_version(version1: str, version2: str) -> int:
    v1 = list(map(int, version1.split('.')))
    v2 = list(map(int, version2.split('.')))

    max_len = max(len(v1), len(v2))
    v1.extend([0] * (max_len - len(v1)))
    v2.extend([0] * (max_len - len(v2)))

    if v1 < v2:
        return -1
    elif v1 > v2:
        return 1
    else:
        return 0
```

```python
def decode_ways(s: str) -> int:
    if not s or s[0] == '0':
        return 0

    n = len(s)
    dp = [0] * (n + 1)
    dp[0] = 1
    dp[1] = 1

    for i in range(2, n + 1):
        if s[i-1] != '0':
            dp[i] += dp[i-1]

        two_digit = int(s[i-2:i])
        if 10 <= two_digit <= 26:
            dp[i] += dp[i-2]

    return dp[n]
```

```python
def title_to_number(column: str) -> int:
    result = 0
    for char in column:
        result = result * 26 + (ord(char) - ord('A') + 1)
    return result
```

```python
def count_and_say(n: int) -> str:
    s = '1'
    for _ in range(n - 1):
        next_s = ''
        i = 0
        while i < len(s):
            char = s[i]
            count = 1
            while i + count < len(s) and s[i + count] == char:
                count += 1
            next_s += str(count) + char
            i += count
        s = next_s
    return s
```

```python
def int_to_roman(n: int) -> str:
    val = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1]
    syms = ['M', 'CM', 'D', 'CD', 'C', 'XC', 'L', 'XL', 'X', 'IX', 'V', 'IV', 'I']

    roman_num = ''
    i = 0
    while n > 0:
        for _ in range(n // val[i]):
            roman_num += syms[i]
            n -= val[i]
        i += 1

    return roman_num
```
